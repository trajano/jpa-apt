3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.regex.Matcher;
6   import java.util.regex.Pattern;
7   
8   import javax.persistence.NamedQuery;
9   
10  public class MetaNamedQuery {
11      private final String methodName;
12      private final String name;
13      private final List<String> parameters;
14  
15      public MetaNamedQuery(final NamedQuery namedQueryAnnotation) {
16          name = namedQueryAnnotation.name();
17          methodName = name.substring(name.lastIndexOf('.') + 1);
18          parameters = new ArrayList<String>();
19          final Matcher m = Pattern.compile("\\:([^=<>\\s\\']+)").matcher(
20                  namedQueryAnnotation.query());
21          while (m.find()) {
22              parameters.add(m.group(1));
23          }
24      }
25  
26      public String getMethodName() {
27          return methodName;
28      }
29  
30      public String getName() {
31          return name;
32      }
33  
34      public List<String> getParameters() {
35          return parameters;
36      }
37  
38  }
