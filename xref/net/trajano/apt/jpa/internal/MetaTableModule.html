17  /**
18   * Represents the table module class that would get generated.
19   *
20   * @author Archimedes Trajano
21   */
22  public class MetaTableModule {
23      /**
24       * Pluralizes a string. The logic is add "s" if the last character is not an
25       * "s" with the following exceptions:
26       * <ul>
27       * <li>add "es" if the input string ends with an "s".
28       * <li>add "es" if the input string ends with an "x".
29       * <li>replace the last character and add "ies" if the input string ends
30       * with an "y".
31       * </ul>
32       *
33       * @param s
34       *            string to pluralize
35       * @return plural string
36       */
37      public static String pluralize(final String s) {
38          if (s.endsWith("s") || s.endsWith("x")) {
39              return s + "es";
40          } else if (s.endsWith("y")) {
41              return s.substring(0, s.length() - 1) + "ies";
42          } else {
43              return s + "s";
44          }
45      }

47      /**
48       * Class name for the table module.
49       */
50      private final String className;
51  
52      /**
53       * Class name of the source entity.
54       */
55      private final String entityClassName;
56  
57      /**
58       * Extra operations.
59       */
60      private final List<MetaOperation> extraOperations;
61  
62      /**
63       * Type for the @Id field.
64       */
65      private final String idType;
66  
67      /**
68       * Named queries.
69       */
70      private final List<MetaNamedQuery> namedQueries;
71  
72      /**
73       * Package name.
74       */
75      private final String packageName;
76  
77      /**
78       * Qualified name of the Table Module class.
79       */
80      private final String qualifiedName;
81  
82      public MetaTableModule(final TypeElement entityType) {
83          packageName = ((PackageElement) entityType.getEnclosingElement())
84                  .getQualifiedName().toString();
85          entityClassName = entityType.getSimpleName().toString();
86          className = pluralize(entityClassName);
87          qualifiedName = pluralize(entityType.getQualifiedName().toString());
88          namedQueries = new ArrayList<MetaNamedQuery>();
89          final NamedQueries namedQueriesAnnotation = entityType
90                  .getAnnotation(NamedQueries.class);
91          if (namedQueriesAnnotation != null) {
92              for (final NamedQuery namedQueryAnnotation : namedQueriesAnnotation
93                      .value()) {
94                  namedQueries.add(new MetaNamedQuery(namedQueryAnnotation));
95              }
96          }
97  
98          String computedIdType = null;
99          extraOperations = new ArrayList<MetaOperation>();
100         for (final Element element : entityType.getEnclosedElements()) {
101             if (ElementKind.FIELD == element.getKind()
102                     && element.getAnnotation(Id.class) != null) {
103                 computedIdType = ((VariableElement) element).asType()
104                         .toString();
105             } else if (isExtraOperation(element)) {
106                 extraOperations.add(new MetaOperation(
107                         (ExecutableElement) element));
108             }
109         }
110         idType = computedIdType;
111     }

141     /**
142      * Checks if the element represents an extra operation. An extra operation
143      * is defined as a public static method that takes in an EntityManager as
144      * it's first parameter.
145      *
146      * @param element
147      *            element to evaluate
148      * @return <code>true</code> if the element represents an extra operation.
149      */
150     private boolean isExtraOperation(final Element element) {
151         if (!ElementKind.METHOD.equals(element.getKind())) {
152             return false;
153         }
154         final ExecutableElement methodElement = (ExecutableElement) element;
155         if (!methodElement.getModifiers().contains(Modifier.PUBLIC)) {
156             return false;
157         }
158         if (!methodElement.getModifiers().contains(Modifier.STATIC)) {
159             return false;
160         }
161         if (!"javax.persistence.EntityManager".equals(methodElement
162                 .getParameters().get(0).asType().toString())) {
163             return false;
164         }
165         return true;
166     }
