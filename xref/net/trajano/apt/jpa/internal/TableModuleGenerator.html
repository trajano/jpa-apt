21  		 for (final MetaNamedQuery namedQuery : meta.getNamedQueries()) {

38  		builder.append("(");
39  		  for (int i = 0 ; i < namedQuery.getParameters().size(); ++i) {

65  		  for (int i = 0 ; i < operation.getParameterNames().size(); ++i) { 
66  		builder.append("     * @param ");

69  		builder.append(operation.getParameterNames().get(i));

80  		builder.append(operation.getParameterDeclarations().get(i));

82  		builder.append(",");
83  		 } 
84  		  } 
85  		builder.append(")\r\n");
86  		  if (!operation.getThrownTypes().isEmpty()) {
87  		builder.append("      throws ");
88  		  for (int i = 0 ; i < operation.getThrownTypes().size(); ++i) {

92  		 }

109 		builder.append(");\r\n    }\r\n\r\n");
